  🚀 Quick Setup Guide

  1. Create Podman Secrets

  # Restic repository password (encrypts backups)
  openssl rand -base64 32 | podman secret create restic-repository-password -

  # SSH key for SFTP access to NAS
  ssh-keygen -t ed25519 -f ~/.ssh/restic_nas_key -C "restic-backup"
  podman secret create restic-ssh-key ~/.ssh/restic_nas_key

  # Add public key to your NAS
  cat ~/.ssh/restic_nas_key.pub
  # Copy this to your NAS user's ~/.ssh/authorized_keys

  # Discord webhook (for notifications)
  echo "https://discord.com/api/webhooks/YOUR_WEBHOOK" | podman secret create restic-discord-webhook -
  echo "YOUR_DISCORD_USER_ID" | podman secret create discord-user-id -

  2. Configure Restic

  # Copy template and edit
  cp configs/restic/restic.env.template configs/restic/restic.env
  nano configs/restic/restic.env

  # Example SFTP configuration:
  RESTIC_REPOSITORY=sftp:backup-user@192.168.1.100:/volume1/backups/homelab-restic

  3. Generate SSH Known Hosts (for SFTP)

  ssh-keyscan -H your-nas-ip > configs/restic/known_hosts

  4. Deploy and Enable

  # Deploy the Quadlet and timer
  ./scripts/deploy.sh

  # Enable automatic daily backups
  systemctl --user enable --now restic-backup.timer

  # Check timer status
  systemctl --user list-timers | grep restic

  5. Test Manually

  # Run first backup manually
  systemctl --user start restic-backup.service

  # Watch logs in real-time
  journalctl --user -u restic-backup.service -f
